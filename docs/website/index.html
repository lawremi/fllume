from fllume import Agent
from pydantic import BaseModel

# Define a structured output format
class StandardizedAddress(BaseModel):
    street: str
    city: str
    state: str
    zip_code: str

# Create an agent to standardize addresses
agent = (Agent.builder()
    .with_model("openai/gpt-4o-mini")
    .with_response_format(StandardizedAddress)
    .build())

# Standardize unstructured data
address = agent.complete(
    "Standardize: 1600 Amphitheatre Pkwy, Mountain View, CA"
)
# address -> StandardizedAddress(...)

See It In Action

from fllume import Agent

# Give the agent a custom tool
def calculate_average(numbers: list[float]) -> float:
    """Calculates the average of a list of numbers."""
    return sum(numbers) / len(numbers)

# Create an agent that can use the tool
agent = (Agent.builder()
    .with_model("openai/gpt-4o-mini")
    .with_tools([calculate_average])
    .build())

# Ask the agent to use the tool
data = [10.5, 20.3, 15.7, 25.1, 12.9]
result = agent.complete(f"What is the average of these numbers: {data}?")
# result -> "The average is 16.9."
