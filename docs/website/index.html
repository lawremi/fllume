See It In Action

from fllume import Agent
from pydantic import BaseModel
import requests

def get_weather(city: str) -> str:
    """Get current weather for a city."""
    # Simplified weather API call
    response = requests.get(f"https://api.weather.com/{city}")
    return response.json()["current"]["condition"]

class WeatherResponse(BaseModel):
    city: str
    condition: str
    recommendation: str

# Create a weather agent
weather_agent = (Agent.builder()
    .with_model("openai/gpt-4")
    .with_instructions("You are a helpful weather assistant")
    .with_tools([get_weather])
    .with_response_format(WeatherResponse)
    .build())

# Get structured weather information
result = weather_agent.complete("What's the weather in Paris and should I bring an umbrella?")
print(f"City: {result.city}")
print(f"Condition: {result.condition}")
print(f"Recommendation: {result.recommendation}")
